package collections;
import java.util.LinkedList;

public class LinkedListExample {

	public static void main(String[] args) {
		LinkedList<String> data = new  LinkedList<>();
		data.add("Hari");
		data.add("mani");
		data.add("Geetha");
		
	for (String d : data) {
			System.out.println(d);
		}

	}

}
